INCLUDE(TribitsAddExecutable)

SEACAS_PACKAGE(Exomatlab)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}"
)

SET(HEADERS "")
APPEND_GLOB(SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.C)

TRIBITS_ADD_EXECUTABLE(
  exomatlab
  NOEXEPREFIX
  NOEXESUFFIX
  SOURCES ${SOURCES}
  COMM serial mpi
  )
install_executable(exomatlab)

SEACAS_POSTPROCESS()
